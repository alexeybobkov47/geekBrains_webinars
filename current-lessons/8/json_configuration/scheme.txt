Arrays:
[значение_1, значение_2, ... , значение_N]

Objects:
{имя_1:значение_1, имя_2:значение_2, ... , имя_N:значение_N}

{
    имя_1:значение_1,
    имя_2:{
        имя_2_1:значение_2_1,
        имя_2_2:значение_2_2,
        }
}

Literally:
{имя_1:"это строка символов", имя_2:10, имя_3:false, имя_4:true, имя_5:null}
